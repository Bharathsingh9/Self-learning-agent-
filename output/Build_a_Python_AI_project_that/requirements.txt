# sentiment_analyzer.py
import tkinter as tk
from tkinter import ttk
from nltk.sentiment.vader import SentimentIntensityAnalyzer
import nltk
import torch
import torch.nn as nn
from transformers import pipeline
from nltk.tokenize import word_tokenize
from nltk.corpus import stopwords
from nltk.stem import PorterStemmer
from nltk.stem import WordNetLemmatizer
import random
import string

nltk.download('vader_lexicon')
nltk.download('punkt')
nltk.download('stopwords')
nltk.download('wordnet')

class SentimentAnalyzer(tk.Frame):
    def __init__(self, master=None):
        super().__init__(master)
        self.master = master
        self.pack()
        self.create_widgets()
        self.sentiment_pipeline = pipeline('sentiment-analysis')

    def create_widgets(self):
        self.text_label = tk.Label(self)
        self.text_label["text"] = "Enter the text you want to analyze:"
        self.text_label.pack(side="top")

        self.text_entry = tk.Text(self)
        self.text_entry.pack(side="top")

        self.analyze_button = tk.Button(self)
        self.analyze_button["text"] = "Analyze"
        self.analyze_button["command"] = self.analyze_sentiment
        self.analyze_button.pack(side="top")

        self.result_label = tk.Label(self)
        self.result_label["text"] = ""
        self.result_label.pack(side="top")

    def analyze_sentiment(self):
        text = self.text_entry.get("1.0", "end-1c")
        text = self.preprocessing(text)
        result = self.sentiment_pipeline(text)
        result = f"{result[0]['label']}: {result[0]['score']}"
        self.result_label["text"] = result

    def preprocessing(self, text):
        lemmatizer = WordNetLemmatizer()
        stemming = PorterStemmer()
        stop_words = set(stopwords.words('english'))
        word_tokens = word_tokenize(text)
        filtered_text = [word for word in word_tokens if word.isalpha()]
        removed_stop_words = [word for word in filtered_text if word.lower() not in stop_words]
        stemmed_text = [stemming.stem(word) for word in removed_stop_words]
        lemmatized_text = [lemmatizer.lemmatize(word) for word in stemmed_text]
        joined_text = ' '.join(lemmatized_text)
        return joined_text

def main():
    root = tk.Tk()
    root.title("Sentiment Analyzer")
    sentiment_analyzer = SentimentAnalyzer(master=root)
    sentiment_analyzer.mainloop()

if __name__ == "__main__":
    main()
