// script.js let displayValue = "0"; let previousValue = "0"; let currentOperation = ""; document.addEventListener("DOMContentLoaded", function () { updateDisplay(); const buttons = document.querySelectorAll("#buttons button"); buttons.forEach((button) => { button.addEventListener("click", () => { const className = button.className; const value = button.textContent; handleButtonPress(className, value); }); }); }); function updateDisplay() { document.getElementById("display").innerHTML = displayValue; } function handleButtonPress(className, value) { switch (className) { case "digit": handleDigitPress(value); break; case "operator": handleOperatorPress(value); break; case "operator=C": handleClearPress(); break; default: break; } } function handleDigitPress(value) { if (displayValue === "0") { displayValue = value; } else { displayValue += value; } updateDisplay(); } function handleOperatorPress(value) { if (currentOperation !== "") { const result = calculateResult(); displayValue = result; } currentOperation = value; previousValue = displayValue; displayValue = ""; updateDisplay(); } function handleClearPress() { displayValue = "0"; currentOperation = ""; previousValue = "0"; updateDisplay(); } function calculateResult() { let result; switch (currentOperation) { case "+": result = parseFloat(previousValue) + parseFloat(displayValue); break; case "-": result = parseFloat(previousValue) - parseFloat(displayValue); break; case "*": result = parseFloat(previousValue) * parseFloat(displayValue); break; case "/": result = parseFloat(previousValue) / parseFloat(displayValue); break; default: break; } return result.toPrecision(10); } function evaluateExpression() { try { const expression = displayValue; const result = eval(expression); displayValue = result.toPrecision(10); updateDisplay(); currentOperation = ""; previousValue = displayValue; } catch (error) { displayValue = "Error"; updateDisplay(); currentOperation = ""; previousValue = displayValue; } }
